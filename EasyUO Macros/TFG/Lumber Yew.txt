sub lumber
comida:
finditem %Comida
if #findkind = -1
{
ignoreitem reset
event sysmessage Comida nao encontrada!
}
GOSUB gs
if #findkind = 1
{
ignoreitem #findid
goto comida
}
GOSUB gs
set #lobjectid #findid
event macro 17 0
GOSUB gs
Set %Chance 0
inicio:
if %Chance >= %Tentativa
{
return
}
GOSUB gs
Set %Chance %Chance + 1

machado:
GOSUB gs
finditem %Machado C
if #findkind = -1
{
ignoreitem reset
event sysmessage Machado nao encontrado!
goto machado
}
GOSUB gs
if #findkind = 1
{
ignoreitem #findid
goto machado
}
GOSUB gs
set #lobjectid #findid
event macro 17 0
GOSUB gs

set #lTargetX %1
set #lTargetY %2
set #lTargetZ 0
set #lTargetKind 3
set #lTargetTile %3
GOSUB gs
target
event macro 25 0
event macro 22 0
Set #SCNT 0

delete journal
scanjournal

if #hits < #maxhits
{
msg Guards ! $
msg Sai fora pau no cu ! $
}

{
msg Guards ! $
msg Sai fora pau no cu ! $
}
if ATTEMPTING_TO_STEAL in #journal
{
msg Guards ! $
msg Sai fora pau no cu ! $
}

{
msg Guards ! $
msg Sai fora pau no cu ! $

if ATTACKING_YOU in #journal
{
msg Guards ! $
msg Sai fora pau no cu ! $
