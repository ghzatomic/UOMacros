sub PegarMaterial
finditem %material g_3
if #findkind = -1
{
setuotitle sem material... em #time
shutdown
}
event drag #findid
wait 20
msg %1 $
wait 20
click %charx %chary
wait 30
return

sub bagsitem
bi:
finditem %bagsid G_3
if #findkind = -1
{
ignoreitem reset
gosub pegarbags
}
if #findbagid = %bag
{
ignoreitem #findid
goto bi
}
set #lobjectid #findid
wait 20
event macro 17 0
wait 20
contpos 606 474
wait 20
ignoreitem reset
bi2:
finditem %item C_ , %bag
if #findkind = -1
{
return
}
set %citem %citem + 1
if %citem > 199
{
set %citem 0
gosub guardarbags
wait 30
return
}
event sysmessage Item de numero %citem sendo guardado!
event drag #findid
wait 10
click 689 558
wait 10
goto bi2



Sub guardarbags
gb:
finditem %bagsid G_3
if #findkind = -1
{
ignoreitem reset
return
}
if #findbagid = %bag
{
ignoreitem #findid
goto gb
}
event drag #findid
wait 30
click %bbagsx %bbagsy
wait 30
ignoreitem reset
return


sub PegarBags
finditem %baubags G_3
if #findkind = -1
{
halt
}
set #lobjectid #findid
wait 10
event macro 17 0
wait 10
finditem %bagsid C_ , %baubags
if #findkind = -1
{
halt
}
event drag #findid
wait 20
click %bgrondx %bgrondy
wait 20
return
